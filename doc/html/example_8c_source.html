211 unsigned int example_get_ag_info(void *w, unsigned int x, unsigned int y) {
212 
213  /* The agent information to return. */
214  unsigned int ag_info = 0;
215 
216  /* Convert generic pointer to world to a WORLD object. */
217  WORLD *my_world = (WORLD *) w;
218 
219  /* Check if the given (x,y) coordinates are within bounds of the world. */
220  if ((x >= my_world->xsize) || (y >= my_world->ysize)) {
221 
222  /* If we got here, then the coordinates are off bounds. As such we will
223  report that the requested agent is of unknown type. No need to
224  specify agent ID or playable status, since the agent is unknown. */
225  ag_info = Unknown;
226 
227  } else {
228 
229  /* Given coordinates are within bounds, let's get and pack the request
230  agent information. */
231 
232  /* Obtain agent at specified coordinates. */
233  AGENT ag = my_world->grid[x * my_world->ysize + y];
234 
235  /* Is there an agent at (x,y)? */
236  if (ag.type == None) {
237 
238  /* If there is no agent at the (x,y) coordinates, set agent type to
239  None. No need to specify agent ID or playable status, since
240  there is no agent here. */
241  ag_info = None;
242 
243  } else {
244 
245  /* If we get here it's because there is an agent at (x,y). Bit-pack
246  all the agent information as specified by the get_agent_info_at
247  function pointer definition. */
248  ag_info = (ag.id << 3) | (ag.playable << 2) | ag.type;
249 
250  }
251 
252  }
253  /* Return the requested agent information. */
254  return ag_info;
255 }
