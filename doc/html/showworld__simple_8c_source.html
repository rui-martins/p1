70 SHOWWORLD *showworld_new(
71  unsigned int xdim,
72  unsigned int ydim,
73  get_agent_info_at aginfo_func) {
74 
75  /* Initialize allegro's functions and addons */
76  al_init();
77  al_init_image_addon();
78  al_init_font_addon();
79  al_init_ttf_addon();
80 
81  SHOWWORLD *sw = NULL;
82  sw = malloc(sizeof(SHOWWORLD));
83  sw->xdim = xdim;
84  sw->ydim = ydim;
85  sw->aginfo_func = aginfo_func;
86 
87  /* Define agent width and height with the proportions to fit on
88  a 600 by 600 display */
89  agent_width = 600/(sw->xdim);
90  agent_height = 600/(sw->ydim);
91 
92  /* Initialize the window display with a fixed size of 600 by 600 */
93  if((sw->xdim) == (sw->ydim)) {
94  display = al_create_display(600,600);
95  }
96  /* Initialize the window display with a width of 600 and
97  a height that maintains the proportions of the objects that'll be drew */
98  if((sw->xdim) > (sw->ydim)) {
99  display = al_create_display((sw->xdim)*agent_width,
100  (sw->ydim)*agent_width);
101  }
102  /* Initialize the window display with a height of 600 and
103  a width that maintains the proportions of the objects that'll be drew */
104  if((sw->xdim) < (sw->ydim)) {
105  display = al_create_display((sw->xdim)*agent_height,
106  (sw->ydim)*agent_height);
107  }
108 
109  /* We do this so that the display draws all of the agents as squares */
110  if((sw->xdim) > (sw->ydim)) {
111  agent_height = 600/(sw->xdim);
112  } else {
113  agent_width = 600/(sw->ydim);
114  }
115 
116  /* Load the respective images of each agent */
117  h_playable = al_load_bitmap("Human1.png");
118  h_ai = al_load_bitmap("Human2.png");
119  z_playable = al_load_bitmap("Zombie1.png");
120  z_ai = al_load_bitmap("Zombie2.png");
121  none = al_load_bitmap("Bg.png");
122 
123  /* Load the fonts that are going to be used and their respective size */
124  if((sw->xdim) >= (sw->ydim)) {
125  font1 = al_load_font("uni0553.ttf",600/((sw->xdim)*3),0);
126  } else {
127  font1 = al_load_font("uni0553.ttf",600/((sw->ydim)*3),0);
128  }
129 
130  /* Clear the background's color (just a precaution) */
131  al_clear_to_color(al_map_rgb(0,0,0));
132 
133  return sw;

152 /* Update the simulation world display/visualization.
153  *
154  * This function obeys the `showworld_update()` prototype defined in
155  * `showworld.h`. */
156 void showworld_update(SHOWWORLD *sw, void *w) {
157 
159  for (unsigned int y = 0; y < sw->ydim; ++y) {
160 
162  for (unsigned int x = 0; x < sw->xdim; ++x) {
163 
166  unsigned int item = sw->aginfo_func(w, x, y);
167 
169  AGENT_TYPE ag_type = item & 0x3;
171  unsigned char playable = (item >> 2) & 0x1;
173  unsigned short ag_id = (item >> 3) & 0xFFFF;
174 
175  /* Determine the agent type. */
176  switch (ag_type) {
177 
178  /* If no agent is present at (x,y), draw Bg. */
179  case None:
180  al_draw_scaled_bitmap(none, 0, 0, 32, 32,
181  x*agent_width, y*agent_height,
182  agent_width, agent_height, 0);
183  break;
184 
185  /* If human agent present at (x,y) draw Human1 or Human2. */
186  case Human:
187  if (playable) {
188  /* Human1 for player-controlled human agent. */
189  al_draw_scaled_bitmap(h_playable, 0, 0, 32, 32,
190  x*agent_width, y*agent_height,
191  agent_width, agent_height, 0);
192  } else {
193  /* Human2 for AI-controlled human agent. */
194  al_draw_scaled_bitmap(h_ai, 0, 0, 32, 32,
195  x*agent_width, y*agent_height,
196  agent_width, agent_height, 0);
197  }
198  /* Print the agent ID in front of it's image. */
199  al_draw_textf(font1, al_map_rgb(255,255,255),
200  x*agent_width, y*agent_height,
201  0, "%i", ag_id);
202  break;
203 
204  /* If zombie agent present at (x,y) draw Zombie1 or Zombie2. */
205  case Zombie:
206  if (playable) {
207  /* Zombie1 for player-controlled zombie agent. */
208  al_draw_scaled_bitmap(z_playable, 0, 0, 32, 32,
209  x*agent_width, y*agent_height,
210  agent_width, agent_height, 0);
211  } else {
212  /* Zombie2 for AI-controlled zombie agent. */
213  al_draw_scaled_bitmap(z_ai, 0, 0, 32, 32,
214  x*agent_width, y*agent_height,
215  agent_width, agent_height, 0);
216  }
217  /* Print the agent ID in front of the image. */
218  al_draw_textf(font1, al_map_rgb(200,0,0),
219  x*agent_width, y*agent_height,
220  0, "%i", ag_id);
221  break;
222 
223  /* Print '?' if unknown type detected. This should *never*
224  happen. */
225  default:
226  al_draw_text(font1, al_map_rgb(255,255,255),
227  x*agent_width, y*agent_height,
228  0, "?");
229 
230  }
231  }
232  }
233  /* We call this function so that the objects can be seen on the
234  display, without it the display would be black*/
235  al_flip_display();
236 }
